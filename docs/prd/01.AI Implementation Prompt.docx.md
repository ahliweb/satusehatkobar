### HR-03: Setiap Route Handler Wajib Melalui Chain Ini

```typescript
// Template wajib untuk setiap route handler
export const handler = async (req: Request, ctx: Context) => {
  // 1. Auth check (dilakukan oleh middleware core, bukan handler)
  const user = ctx.user // sudah ter-inject oleh middleware

  // 2. RBAC check
  if (!user.hasPermission('duty.create')) {
    return ctx.forbidden('Anda tidak memiliki izin untuk membuat ST')
  }

  // 3. ABAC check
  if (user.organization_level === 'faskes' && body.health_facility_id !== user.faskes_id) {
    return ctx.forbidden('Anda hanya dapat membuat ST untuk faskes Anda')
  }

  // 4. Input validation
  const result = DutyRequestSchema.safeParse(await req.json())
  if (!result.success) {
    return ctx.validationError(result.error)
  }

  // 5. Business logic via service
  const created = await dutyService.create(result.data, user)

  // 6. Audit log (wajib untuk semua mutasi)
  await auditLogger.emit({
    actor_id: user.id,
    actor_name: user.name,
    plugin_key: 'duty-travel',
    action: 'duty.created',
    entity_type: 'duty_requests',
    entity_id: created.id,
    after_data: created,
    ip_address: req.headers.get('CF-Connecting-IP') ?? '',
  })

  // 7. Response
  return ctx.created({ data: created })
}
```

## 3. Arsitektur Plugin yang Benar

### 3.1 Struktur Folder Plugin

```
plugins/[plugin-name]/
├── package.json            # Plugin manifest dengan plugin_key
├── README.md               # Deskripsi dan setup
├── CHANGELOG.md            # Riwayat perubahan (Keep a Changelog)
├── src/
│   ├── index.ts            # Entry point, route registration
│   ├── routes/             # Route handlers (thin layer)
│   │   └── [entity].routes.ts
│   ├── services/           # Business logic (thick layer, unit tested)
│   │   └── [entity].service.ts
│   ├── repositories/       # DB queries (parameterized only)
│   │   └── [entity].repository.ts
│   ├── schemas/            # Zod validation schemas
│   │   └── [entity].schema.ts
│   ├── types/              # TypeScript type definitions
│   │   └── [entity].types.ts
│   └── errors/             # Custom error classes
│       └── [entity].errors.ts
├── ui/                     # React components (jika ada)
│   └── pages/
├── migrations/             # DB migrations (ordered)
│   └── [YYYYMMDDHHMMSS]_[desc].sql
├── seeds/                  # Seed data
│   └── [entity].seed.sql
└── docs/
    ├── API.md
    ├── SCHEMA.md
    └── PERMISSIONS.md
```

### 3.2 Lapisan Kode (Layered Architecture)

```
HTTP Request
    ↓
[Route Handler] — thin, hanya orchestration
    ↓
[Service Layer] — business logic, ABAC, validation
    ↓
[Repository Layer] — DB queries, parameterized
    ↓
[D1 Database]
```

- **Route Handler**: Hanya auth check, RBAC, dispatch ke service, emit audit, return response
- **Service Layer**: Business rules, ABAC, state validation, orchestration antar repo
- **Repository Layer**: Query DB saja, tidak ada business logic

## 7. Alur Approval yang Benar

### 7.1 Membangun Approval Chain dari Konfigurasi

```typescript
// Saat duty_request disubmit:
async function buildApprovalChain(dutyRequest: DutyRequest, db: D1Database) {
  const orgLevel = dutyRequest.organization_level // 'dinas' atau 'faskes'

  // Load konfigurasi dari DB (bukan hardcoded!)
  const stepConfigs = await db
    .prepare('SELECT * FROM duty_approval_step_config WHERE org_level = ? AND is_active = 1 ORDER BY step_order')
    .bind(orgLevel)
    .all()

  const approvals = []
  for (const config of stepConfigs.results) {
    let status = 'pending'

    // Auto-skip berdasarkan skip_condition
    if (config.skip_condition) {
      const condition = JSON.parse(config.skip_condition)
      if (shouldSkip(dutyRequest, condition)) {
        status = 'skipped'
      }
    }

    approvals.push({
      id: crypto.randomUUID(),
      duty_request_id: dutyRequest.id,
      approval_step: config.approval_step,
      approver_role: config.approver_role,
      sort_order: config.step_order,
      status,
      created_at: new Date().toISOString(),
    })
  }

  return approvals
}

function shouldSkip(request: DutyRequest, condition: Record<string, unknown>): boolean {
  return Object.entries(condition).every(([key, value]) => {
    return (request as Record<string, unknown>)[key] === value
  })
}
```

### 7.2 Validasi Approval Step Aktif

```typescript
async function validateApproverCanAct(dutyRequestId: string, user: User, db: D1Database) {
  // Temukan langkah approval yang sedang pending
  const activeStep = await db
    .prepare(`
      SELECT * FROM duty_request_approvals
      WHERE duty_request_id = ? AND status = 'pending'
      ORDER BY sort_order ASC LIMIT 1
    `)
    .bind(dutyRequestId)
    .first()

  if (!activeStep) {
    throw new WorkflowError('Tidak ada langkah approval yang aktif')
  }

  if (activeStep.approver_role !== user.primary_role) {
    throw new ForbiddenError(
      `Langkah ini membutuhkan role ${activeStep.approver_role}, Anda memiliki role ${user.primary_role}`
    )
  }
}
```

## 9. Anti-Patterns yang Harus Dihindari

| Anti-Pattern | Mengapa Berbahaya | Solusi |
|-------------|------------------|--------|
| Hardcode approval chain | Tidak bisa dikonfigurasi per org | Baca dari `duty_approval_step_config` |
| Direct table join lintas plugin | Coupling yang tidak terkontrol | Service contract API |
| Expose R2 path langsung | Bypass access control | Signed URL atau proxy Worker |
| Simpan PII di KV cache | KV tidak terenkripsi per key | KV hanya untuk agregat/config |
| Skip audit log | Melanggar HR-07 | Selalu emit sebelum return response |
| `console.log` di production | Bocor info sensitif | Gunakan structured logger |
| `SELECT *` tanpa filter | Data leak lintas unit/faskes | Selalu tambahkan ABAC WHERE clause |
| Mutable arsip | Merusak audit integrity | Arsip hanya INSERT, tidak UPDATE pada field inti |
| Status yang tidak valid | Workflow rusak | Validasi status sebelum update via allowedTransitions map |

### 9.1 Status Transition Validation

```typescript
const ALLOWED_DUTY_TRANSITIONS: Record<string, string[]> = {
  'draft': ['submitted', 'cancelled'],
  'submitted': ['supervisor_verified', 'returned', 'rejected'],
  'supervisor_verified': ['technical_verified', 'returned', 'rejected'],
  'technical_verified': ['secretary_verified', 'returned', 'rejected'],
  'secretary_verified': ['finance_verified', 'final_approved', 'returned', 'rejected'],
  'finance_verified': ['final_approved', 'returned', 'rejected'],
  'final_approved': ['document_issued'],
  'document_issued': ['in_progress'],
  'in_progress': ['evidence_submitted'],
  'evidence_submitted': ['evidence_verified', 'in_progress'],
  'evidence_verified': ['completed'],
  'completed': ['archived'],
}

function validateStatusTransition(currentStatus: string, newStatus: string): void {
  const allowed = ALLOWED_DUTY_TRANSITIONS[currentStatus] ?? []
  if (!allowed.includes(newStatus)) {
    throw new WorkflowError(
      `Transisi status dari '${currentStatus}' ke '${newStatus}' tidak diizinkan`
    )
  }
}
```
